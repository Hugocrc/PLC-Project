aritmética:
sum(x, y) # x+y
subtrac(x, y) # x-y
mult(x, y) # x*y
div(x, y) # x / y
rem(x, y) # resto da div de x por y

relacionais:
equals(x, y) # x == y
gt(x,y) # x > y
gte(x, y) # x >= y
lt(x, y) # x < y
lte(x, y) # x <= y
notequals(x, y) # x != y

lógicas:
not(x) # negação de x
and(x, y) # conjunção de x com y
or(x, y) # disjunção de x com y

STDIN/STDOUT:
write(x) # escreve x no stdout
read() # lê algo do stdin

ciclos:

end;
